  <meta name="viewport" content="width=device-width, initial-scale=1.0">
</head>
<body style="font-family: Arial, sans-serif; text-align:center; margin-top:50px;">
  <h2>QR Mission Scanner</h2>
  
  <label>Your Name:</label><br>
  <input type="text" id="name" placeholder="Enter your name"><br><br>

  <label>Scanned Code:</label><br>
  <input type="text" id="code" placeholder="Enter code"><br><br>

  <button onclick="submitScan()">Submit</button>

  <p id="status" style="margin-top:20px; font-weight:bold;"></p>

  <script>
    const BACKEND_URL = "https://script.google.com/macros/s/AKfycbz8HkwVcoOA1M__4b3Oxv9HXekPbIUiXP7aRm01FYSTY7RkIAbTB0uq5-n0x7eK0O7iTQ/exec"; // replace with Apps Script web app link

    async function submitScan() {
      const name = document.getElementById("name").value.trim();
      const code = document.getElementById("code").value.trim();
      if (!name || !code) {
        alert("Please enter both name and code");
        return;
      }

      document.getElementById("status").innerText = "Submitting...";

      try {
        const response = await fetch(BACKEND_URL, {
          method: "POST",
          body: JSON.stringify({name, code}),
          headers: {"Content-Type": "application/json"}
        });
        const text = await response.text();
        document.getElementById("status").innerText = text;
        document.getElementById("code").value = "";
      } catch (err) {
        document.getElementById("status").innerText = "Error: " + err;
      }
    }

    // If QR links have ?code=XXX, auto-fill the code field
    window.onload = () => {
      const params = new URLSearchParams(window.location.search);
      if (params.has("code")) {
        document.getElementById("code").value = params.get("code");
      }
    };
  </script>
</body>
